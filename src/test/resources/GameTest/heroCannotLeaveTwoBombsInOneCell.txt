
before()

        ☼☼☼☼☼
        ☼   ☼
        ☼ x ☼
        ☼ ☺ ☼
        ☼☼☼☼☼,

        ☼☼☼☼☼
        ☼   ☼
        ☼ X ☼
        ☼   ☼
        ☼☼☼☼☼,
        0)

        ☼☼☼☼☼
        ☼   ☼
        ☼☺  ☼
        ☼   ☼
        ☼☼☼☼☼,
